Table users {
  user_id integer [primary key]
  username varchar
  email varchar
  timezone varchar
  created_at timestamp
}

Table calendars {
  calendar_id integer [primary key]
  user_id integer
  name varchar
  color_code varchar
}

Table events {
  event_id integer [primary key]
  calendar_id integer
  title varchar
  description text
  start_time timestamp
  end_time timestamp
  location varchar
  is_all_day boolean
}

Table event_attendees {
  event_id integer
  user_id integer
  rsvp_status varchar
}

Table reminders {
  reminder_id integer [primary key]
  event_id integer
  reminder_type varchar
  minutes_before integer
}

Ref: calendars.user_id > users.user_id
Ref: events.calendar_id > calendars.calendar_id
Ref: event_attendees.event_id > events.event_id
Ref: event_attendees.user_id > users.user_id
Ref: reminders.event_id > events.event_id
