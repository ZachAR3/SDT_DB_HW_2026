Table users {
  id integer [primary key]
  username varchar
  email varchar
  created_at timestamp
}

Table events {
  id integer [primary key]
  title varchar
  start_time timestamp
  end_time timestamp
  owner_id integer
}

Table event_participants {
  event_id integer
  user_id integer
  status varchar // e.g., 'going', 'maybe', 'declined'
}

Ref: events.owner_id > users.id
Ref: event_participants.event_id > events.id
Ref: event_participants.user_id > users.id
